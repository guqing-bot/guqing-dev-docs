每个 Poetora 站点都需要一个包含核心配置设置的 `docs.json` 文件。了解更多关于下面的[属性](#属性)。

## 属性

<ResponseField name="name" type="string" required>
您项目的名称。用于全局标题。

示例：`poetora`

</ResponseField>

<ResponseField name="navigation" type="Navigation[]" required>
  包含该组内所有页面的组数组
  <Expandable title="导航">
    <ResponseField name="group" type="string">
    组的名称。

    示例：`设置`

    </ResponseField>
    <ResponseField name="pages" type="string[]">
    将作为页面的 markdown 文件的相对路径。

    示例：`["customization", "page"]`

    </ResponseField>

  </Expandable>
</ResponseField>

<ResponseField name="logo" type="string or object">
  徽标图像的路径或包含"浅色"和"深色"模式徽标图像路径的对象
  <Expandable title="徽标">
    <ResponseField name="light" type="string">
      浅色模式下的徽标路径
    </ResponseField>
    <ResponseField name="dark" type="string">
      深色模式下的徽标路径
    </ResponseField>
    <ResponseField name="href" type="string" default="/">
      点击徽标链接到的位置
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="favicon" type="string">
  网站图标图像的路径
</ResponseField>

<ResponseField name="colors" type="Colors">
  全局主题的十六进制颜色代码
  <Expandable title="颜色">
    <ResponseField name="primary" type="string" required>
      主要颜色。在浅色模式下最常用于突出显示内容、章节标题、重点内容
    </ResponseField>
    <ResponseField name="light" type="string">
      深色模式的主要颜色。在深色模式下最常用于突出显示内容、章节标题、重点内容
    </ResponseField>
    <ResponseField name="dark" type="string">
      重要按钮的主要颜色
    </ResponseField>
    <ResponseField name="background" type="object">
      浅色和深色模式下背景的颜色
      <Expandable title="对象">
        <ResponseField name="light" type="string" required>
          浅色模式下背景的十六进制颜色代码
        </ResponseField>
        <ResponseField name="dark" type="string" required>
          深色模式下背景的十六进制颜色代码
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarLinks" type="TopbarLink[]">
  您要在顶部栏中包含的链接的 `name` 和 `url` 数组
  <Expandable title="顶部栏链接">
    <ResponseField name="name" type="string">
    按钮的名称。

    示例：`联系我们`
    </ResponseField>
    <ResponseField name="url" type="string">
    点击按钮后的链接。示例：`https://poetora.com/docs`
    </ResponseField>

  </Expandable>
</ResponseField>

<ResponseField name="topbarCtaButton" type="Call to Action">
  <Expandable title="顶部栏行动号召">
    <ResponseField name="type" type={'"link" or "github"'} default="link">
    链接显示按钮。GitHub 显示提供的 URL 处的存储库信息，包括 GitHub 星数。
    </ResponseField>
    <ResponseField name="url" type="string">
    如果是 `link`：按钮链接到的位置。
    
    如果是 `github`：要从中加载 GitHub 信息的存储库链接。
    </ResponseField>
    <ResponseField name="name" type="string">
    按钮内的文本。仅当 `type` 是 `link` 时才需要。
    </ResponseField>

  </Expandable>
</ResponseField>

<ResponseField name="versions" type="string[]">
  版本名称数组。仅当您想在导航栏中使用下拉菜单显示不同版本的文档时才使用此选项。
</ResponseField>

<ResponseField name="anchors" type="Anchor[]">
  锚点数组，包括 `icon`、`color` 和 `url`。
  <Expandable title="锚点">
    <ResponseField name="icon" type="string">
    用于突出显示锚点的 [Font Awesome](https://fontawesome.com/search?q=heart) 图标。

    示例：`comments`
    </ResponseField>
    <ResponseField name="name" type="string">
    锚点标签的名称。

    示例：`社区`
    </ResponseField>
    <ResponseField name="url" type="string">
      标记哪些页面进入锚点的 URL 开头。通常，这是您放置页面的文件夹的名称。
    </ResponseField>
    <ResponseField name="color" type="string">
      锚点图标背景的十六进制颜色。如果您传递具有 `from` 和 `to` 属性的对象，也可以是渐变色，每个属性都是十六进制颜色。
    </ResponseField>
    <ResponseField name="version" type="string">
      如果您想隐藏锚点直到选择正确的文档版本时使用。
    </ResponseField>
    <ResponseField name="isDefaultHidden" type="boolean" default="false">
      如果您想隐藏锚点直到您直接链接某人到其中的文档，则传递 `true`。
    </ResponseField>
    <ResponseField name="iconType" default="duotone" type="string">
      以下之一："brands"、"duotone"、"light"、"sharp-solid"、"solid" 或 "thin"
    </ResponseField>

  </Expandable>
</ResponseField>

<ResponseField name="topAnchor" type="Object">
  覆盖最顶部锚点的默认配置。
  <Expandable title="对象">
    <ResponseField name="name" default="Documentation" type="string">
      最顶部锚点的名称
    </ResponseField>
    <ResponseField name="icon" default="book-open" type="string">
      Font Awesome 图标。
    </ResponseField>
    <ResponseField name="iconType" default="duotone" type="string">
      以下之一："brands"、"duotone"、"light"、"sharp-solid"、"solid" 或 "thin"
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="tabs" type="Tabs[]">
  导航标签数组。
  <Expandable title="标签">
    <ResponseField name="name" type="string">
      标签标签的名称。
    </ResponseField>
    <ResponseField name="url" type="string">
      标记哪些页面进入标签的 URL 开头。通常，这是您放置页面的文件夹的名称。
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="backgroundImage" type="string">
  要在每个页面后面显示的背景图像。查看 [Infisical](https://infisical.com/docs)
  和 [FRPC](https://frpc.io) 的示例。
</ResponseField>
